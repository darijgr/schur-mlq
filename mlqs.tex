A \defn{flagged Schur function} is
\[
\fs_{\lambda}(b; x_1, \dotsc, x_n) = \sum_T x^T,
\]
where we sum over all flagged semistandard tableaux.

\section{Main result}
\label{sec:result}

We define the weight of an MLQ $M$ with spectral parameters $(x_1, \dotsc, x_N)$ by
\[
\wt(M) = \prod_{(b_1,b_2)} x_{b_1}
\]
where the product is taken over the positions balls.

Let
\[
\mcM_k(b; x_1, \dotsc, x_n) = \sum_M \wt(M).
\]

\begin{thm}
We have
\[
\mcM_k(b; x_1, \dotsc, x_n) = \fs_{\lambda}(b; x_1, \dotsc, x_n).
\]
\end{thm}
